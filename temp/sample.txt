public Mono<List<InventoryOnhandDetails>> getReworkItemsOnhandDetails(String organizationCode) {
    // Step 1: Validate the organizationCode
    if (organizationCode == null || organizationCode.isBlank() || !organizationCode.matches("^[a-zA-Z0-9_-]+$")) {
        return Mono.error(new IllegalArgumentException("Invalid organization code"));
    }

    // Step 2: Sanitize the organizationCode
    organizationCode = organizationCode.trim();

    // Step 3: Proceed with the sanitized and validated organizationCode
    return itemRepository.findReworkItems(organizationCode);
}
